# -*- coding: uft-8 -*-
import requests
import re
def get_data(url, pattern)
	html = request.get(url)
	data = re.findall(pattern, html.text)
	return data

# -*- coding: uft-8 -*-
import requests
import json
def get_data(ajax_url, key)
	html = request.get(ajax_url)
	data = json.loads(ajax_url.context)[key]
	return data
